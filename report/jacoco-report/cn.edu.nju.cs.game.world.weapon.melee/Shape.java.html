public enum Shape implements Serializable {

    Quater("Quater"),

    Half("Half"),

    Full("Full");

    private String shape;
    public String shape(){
        return this.shape;
    }

    /**
     * @author: Xu YangXin
     * @param {String} shape 形状的名字
     * @return {*}
     * @description: 构造器
     */
    Shape(String shape){
        this.shape = shape;
    }
}
