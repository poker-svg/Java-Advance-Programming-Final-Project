package cn.edu.nju.cs.game.world.creature.bros;

import java.awt.Color;

import cn.edu.nju.cs.game.world.item.Tile;
import cn.edu.nju.cs.game.world.item.World;

public class SecondBro extends Player{
    public static final char GLYPH = Tile.SECOND_BRO.glyph();
    public static final Color COLOR = Tile.SECOND_BRO.color();
    public static final int MAX_HP = 100;
    public static final int ATTACK_VALUE = 15;
    public static final int DEFENSE = 5;
    public static final int VISION_RADIUS = 15;
    public static final int MOVE_SPEED_INTERVAL = 500;

    /**
     * @author: Xu YangXin
     * @param {World} world 所处世界
     * @return {*}
     * @description: 构造器
     */    
    public SecondBro(World world){
        super(world, SecondBro.GLYPH, SecondBro.COLOR, SecondBro.MAX_HP, SecondBro.ATTACK_VALUE,
                SecondBro.DEFENSE, SecondBro.VISION_RADIUS, SecondBro.MOVE_SPEED_INTERVAL);

        this.weaponFactory.newSniperGun();
        this.weaponFactory.newBombGlove();
    }
}
